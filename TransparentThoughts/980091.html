WEBrick's Deadlock Bug And Solution- -

                                      

# File: /app/controllers/test_controller
require 'open-uri'

class TestController < ApplicationController
  def test1
    render :text => "Hello, World!"
  end
 
  def test2
    url = url_for :controller => 'test', :action => 'test1'
    result = ''
    open(url) do |http|
      result = http.read
    end
    render :text => result
  end
end
# file end
# when accessing /test/test2 under WEBrick, will deadlock and timeout
