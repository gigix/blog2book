Servlet with Scheme: Say Hello- -

                                      

(require (lib "unitsig.ss")
         (lib "servlet-sig.ss" "web-server"))

;; return value of a servlet unig/sig will be displayed as a web page
(unit/sig () (import servlet^)
 
  (define the-text "Hello, Web!")
  
  ;; x-expression, will generate XML as following:
  ;; <html><head><title>"Hello, Web!"</title></head>
  ;; <body bgcolor="white"><p>"Hello, Web!"</p></body></html>
  `(html (head (title ,the-text))
         (body ([bgcolor "white"])
               (p ,the-text)))

)
